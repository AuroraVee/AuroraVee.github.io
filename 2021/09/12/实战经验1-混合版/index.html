    11.检测DOM更新

    MutationObserver是HTML5新增的属性，用于监听DOM修改事件，能够监听到节点的属性、文本内容、子节点等的改动

    Vue.nextTick可以在DOM更新完毕之后执行一个回调，vue进行DOM更新内部也是调用nextTick来做异步队列控制。而当我们自己调用nextTick的时候，它就在更新DOM的那个microtask后追加了我们自己的回调函数，从而确保我们的代码在DOM更新后执行，同时也避免了setTimeout可能存在的多次执行问题。也就是说VUE是借助MutationObserver这一宏任务特性进行的更新回调

    15.window.devicePixelRatio解决canvas模糊问题

    Window 接口的devicePixelRatio返回当前显示设备的物理像素分辨率与CSS像素分辨率之比。它告诉浏览器应使用多少屏幕实际像素来绘制单个CSS像素。
    $$
    物理像素=CSS像素*devicePixelRatio
    $$

    var canvas = document.getElementById('canvas');

// Set display size (css pixels).
var size = 200;
canvas.style.width = size + "px";
canvas.style.height = size + "px";

// Set actual size in memory (scaled to account for extra pixel density).
var scale = window.devicePixelRatio; // Change to 1 on retina screens to see blurry canvas.
canvas.width = Math.floor(size * scale);
canvas.height = Math.floor(size * scale);

    防抖:

    1. 动作绑定事件，触发完事件后 n 秒内不再触发事件，事件处理程序才执行。要不然就又重新开始计时，始终以最新的触发为准

};
function debounce (func, wait) {
    var timeout

    return function () {
        var context = this //指向container
        var arg = arguments //指向事件处理对象
        clearTimeout(timeout) 
 //如果在wait时间内又触发了事件，则把上一次的定时清除，然后重新计时
 //事件触发的时间至少要间隔wait时间，才会去执行相应的程序
        timeout = setTimeout(function () {
            func.apply(context, arg)
        }, wait)
    }

    ​ 2.触发事件就立即执行函数，但是要间隔wait时间才会重新触发

    //传入immediate=true
function debounce(func, wait, immediate) {

    var timeout, result;

    return function () {
        var context = this;
        var args = arguments;

        if (timeout) clearTimeout(timeout);//清除定时，timeout为定时器ID，这个不会改变
        if (immediate) {
            // 如果已经执行过，不再执行
            var callNow = !timeout;
            timeout = setTimeout(function(){
                timeout = null; 
                //定时到将id置位null，如果定时还没结束又触发了事件，会执行clearTimeout清除定时，callNow为false
            }, wait)
            if (callNow) result = func.apply(context, args) //立即执行了处理函数
        }
        else {
            timeout = setTimeout(function(){
                func.apply(context, args)
            }, wait);
        }
        return result;
    }
}

    ​ 3.通过按钮暂时取消防抖

    function debounce(func, wait, immediate) {

    var timeout, result;

    var debounced = function () {
        var context = this;
        var args = arguments;

        if (timeout) clearTimeout(timeout);
        if (immediate) {
            // 如果已经执行过，不再执行
            var callNow = !timeout;
            timeout = setTimeout(function(){
                timeout = null;
            }, wait)
            if (callNow) result = func.apply(context, args)
        }
        else {
            timeout = setTimeout(function(){
                func.apply(context, args)
            }, wait);
        }
        return result;
    };

    debounced.cancel = function() {
        clearTimeout(timeout); //先清除计时，只是暂时清除防抖，下一次触发仍然有计时限制；如果没有这个，timeout将一直为null
        timeout = null; //点击后callnow为True，所以可以立即执行
    };

    return debounced;
}

    节流:每隔一段时间，执行一次事件处理程序

    1. 时间戳，比较当前触发时间和上一次执行事件处理程序的时间
    function throttle(func, wait) {
    var context, args;
    var previous = 0; // 上一次事件处理程序执行的时间戳

    return function() {
        var now = +new Date(); // 当前事件触发时的时间戳
        context = this;
        args = arguments;
        if (now - previous > wait) {
            func.apply(context, args);
            previous = now;
        }
    }
}
// 鼠标移入时立即执行，鼠标移出时不再执行，中间鼠标不断移动，间隔wait时间执行一次函数

    ​ 2.定时器：

    function throttle(func, wait) {
    var timeout;
    return function() {
        context = this;
        args = arguments;
        if (!timeout) {
            timeout = setTimeout(function(){
                timeout = null;
                func.apply(context, args)
            }, wait)
        }
    }
}
// 鼠标移入时延时了wait才执行函数；间隔wait时间执行函数；如果定时已经开启时不触发事件，也会过一段时间后执行一次函数

    ​ 3.鼠标移入能立刻执行，停止触发的时候还能再执行一次

    function throttle(func, wait) {
    var timeout, context, args, result;
    var previous = 0; // 保存上一次事件处理程序的时间戳

    var later = function() {
        previous = +new Date();
        // 保存上一次事件处理程序的时间戳
        timeout = null;
        func.apply(context, args)
    };

    var throttled = function() {
        var now = +new Date(); // 当前触发的时间戳
        //下次触发 func 剩余的时间
        var remaining = wait - (now - previous);
        context = this;
        args = arguments;
         // 如果没有剩余的时间了或者你改了系统时间
        if (remaining <= 0 || remaining > wait) {
            if (timeout) {
                clearTimeout(timeout);
                timeout = null;
            }
            previous = now;// 保存上一次事件处理程序的时间戳
            func.apply(context, args);
        } else if (!timeout) {
            timeout = setTimeout(later, remaining);
            // 延时剩下的时间去触发
        }
    };
    return throttled;
}

    区别：防抖是虽然事件持续触发，但只有等事件停止触发后 n 秒才执行函数，节流是持续触发的时候，每 n 秒执行一次函数

    18.3.promise.all并发限制

    在瞬间发出几十万个http请求，这样很有可能导致堆积了无数调用栈导致内存溢出。

    Promise.all并发限制指的是，每个时刻并发执行的promise数量是固定的，最终的执行结果还是保持与原来的Promise.all一致。

    18.4.例题

    实现一个批量请求函数 multiRequest(urls, maxNum)，要求如下：
    • 要求最大并发数 maxNum
    • 每当有一个请求返回，就留下一个空位，可以增加新的请求
    • 所有请求完成后，结果按照 urls 里面的顺序依次打出

    function multiRequest(urls = [], maxNum) {
  // 请求总数量
  const len = urls.length;
  // 根据请求数量创建一个数组来保存请求的结果
  const result = new Array(len).fill(false);
  // 当前完成的数量
  let count = 0;

  return new Promise((resolve, reject) => {
    // 请求maxNum个
    while (count < maxNum) {
      next();
    }
    function next() {
      let current = count++;
      // 处理边界条件
      if (current >= len) {
        // 请求全部完成就将promise置为成功状态, 然后将result作为promise值返回
        !result.includes(false) && resolve(result);
        return;
      }
      const url = urls[current];
      console.log(`开始 ${current}`, new Date().toLocaleString());
      fetch(url)//触发异步操作后，程序会继续往下执行，连续调用maxNum个next后，即同时触发了maxnum个异步操作，然后前面完成的操作会通知下一个触发新的异步
        .then((res) => {
          // 保存请求结果
          result[current] = res;
          console.log(`完成 ${current}`, new Date().toLocaleString());
          // 请求没有全部完成, 就递归
          if (current < len) {
            next();
          }
        })
        .catch((err) => {
          console.log(`结束 ${current}`, new Date().toLocaleString());
          result[current] = err;
          // 请求没有全部完成, 就递归
          if (current < len) {
            next();
          }
        });
    }
  });
}

    19.keyup、keypress、keydown事件

    事件触发顺序：keydown -> keypress ->textInput ->keyup

    keydown和keypress事件触发在文字还没敲进文本框，这时如果在keydown和keypress事件中输出文本框中的文本，得到的是触发键盘事件前的文本，而keyup事件触发时整个键盘事件的操作已经完成，获得的是触发键盘事件后的文本

    在keyup事件中无法阻止浏览器默认事件，因为在keypress时，浏览器默认行为已经完成，即将文字输入文本框（尽管这时还没显示），这个时候不管是preventDefault还是return false，都不能阻止在文本框中输入文字的行为。如要阻止在文本框中输入文字，必须在keydown或keypress时阻止

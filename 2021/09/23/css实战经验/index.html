    实战经验

    1.border

    把border-style设置为outset时，右下的border-color颜色会比实际给的颜色值深

    <label style="border:outset 4px rgba(127, 127, 127, 255)">test1</label><br/>
<label style="border:solid 4px rgba(127, 127, 127, 255)">test2</label>

    2.attr()表达式

    CSS表达式 attr() 用来获取选择到的元素的某一HTML属性值，并用于其样式。它也可以用于伪元素，属性值采用伪元素所依附的元素。目前支持的仅有伪元素的 content 属性

    p:before {
    content:attr(data-foo) " ";
}/* 获取p标签的data-foo属性*/

    3.计算canvas上的鼠标位置

    3.1一般情况

    canvas.addEventListener("click", function callBack(evt) {
    //获取鼠标相对于客户端可视区的位置，不随页面滚动变化
    let x = evt.clientX;
    let y = evt.clientY;
    
    //获取canvas相对于客户端可视区的位置，不随页面滚动变化
    let rect = canvas.getBoundingClientRect();
    
    //鼠标相对于canvas的位置
    x -= rect.left;
    y -= rect.top;
    console.log(x, y); // (x, y) 就是鼠标在 canvas 单击时的坐标
});

    3.2 canvas设置了 border/padding

    //获得元素的计算属性
let style = window.getComputedStyle(canvas, null);

let borderLeft = parseFloat(style["border-left-width"]);
let borderTop = parseFloat(style["border-top-width"]);
let paddingLeft = parseFloat(style["padding-left"]);
let paddingTop = parseFloat(style["padding-top"]);

canvas.addEventListener("click", function callBack(evt) {
    let x = evt.clientX;
    let y = evt.clientY;
    
    let rect = canvas.getBoundingClientRect();
    
    x -= rect.left - borderLeft - paddingLeft; // 去除 borderLeft paddingLeft 后的坐标
    y -= rect.top - borderTop - paddingTop; // 去除 borderLeft paddingLeft 后的坐标
    console.log(x, y); // (x, y) 就是鼠标在 canvas 单击时的坐标
});

    3.3 设置了CSS样式中的 width/height

    <canvas
        id="canvas"
        width="300"
        height="300"
        style="
          left: 0px;
          top: 0px;
          position: absolute;
          width: 500px;
          height: 600px;
        "
      ></canvas>
<!-- 同时有canvas宽高和CSS宽高，最终的是显示CSS宽高 -->

    let style = window.getComputedStyle(canvas, null);
let cssWidth = parseFloat(style["width"]);
let cssHeight = parseFloat(style["height"]);

let scaleX = canvas.width / cssWidth; // 水平方向的缩放因子
let scaleY = canvas.height / cssHeight; // 垂直方向的缩放因子
canvas.addEventListener("click", function __handler__(evt) {
    let x = evt.clientX;
    let y = evt.clientY;
    let rect = canvas.getBoundingClientRect();
    x -= rect.left;
    y -= rect.top;
    
    //鼠标在canvas宽高上的位置，范围X:[0,300] Y:[0,300]
    x *= scaleX; // 修正水平方向的坐标
    y *= scaleY; // 修正垂直方向的坐标
    console.log(x, y); // (x, y) canvas 里的坐标
});

    4.getComputedStyle

    该方法会返回一个对象，包含了所有CSS属性的值

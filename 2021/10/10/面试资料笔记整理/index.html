    面试题笔记整理

    注：本笔记涉及到的题来源于「帅地玩编程」公众号提供的程序员内功修炼笔记，原来的笔记是用Java实现的，我采用JavaScript重写了一遍，并且对实现方法做了一些优化。感兴趣的同学可以对比着看一下

    1.必学算法与数据结构

    1.1 时间复杂度和空间复杂度

    时间复杂度排序：O(1) < O(logn) < O(n) < O(nlogn) < O(n^2) < O(n^3) < O(2^n) < O(n!) < O(n^n)

    1.算法总结

    1.1 递归

    1.1.1 递归的三要素

    • 明确递归函数要实现的功能

      即题目要求的值

    • 寻找递归的结束条件

      当函数参数为某个值的时候，能知道函数的结果，通过return结束递归。一维一般是把函数参数缩到最小

    • 找到函数的等价关系式

      不断缩小函数参数的范围，将原来的函数等价为更小的参数表达式。等价变换后需要再次验证第2步的结束条件是否合理，避免出现死循环

    1.1.2 例题

    「例题1」

    求n的阶乘

    //函数功能f(n)：n的阶乘
//结束条件：如果n>=2，那么n=2是结束条件；如果n>0,那么n=1是结束条件；终上结束条件n<=2 return n
//等价关系式：f(n)=n!=n*(n-1)!=n*f(n-1)
function f(n){
    if(n<=2){
        return n
    }
    return n*f(n-1)
}

    1.1.3 优化

    方法一：存储已经计算过的值****（函数记忆）

    每次先从存储变量中查找当前值是否存在，如果存在则直接返回；不存在则将计算后的值保存下来，然后返回。以空间换时间

    //对例题3的代码进行优化

//优化方法：将每次的递归值存储下来，有则直接返回。 
//这个比原作者的优化方法更好，函数调用只有n次
let cacheMap = new Map()

function f1 (n) {

    if (n <= 2) {
        return n
    } else {

        let value1 = cache(cacheMap, n - 1)
        let value2 = cache(cacheMap, n - 2)
        return value1 + value2
    }

    function cache (cacheMap, number) {
        if (!cacheMap.has(number)) {

            let value = f1(number)
            cacheMap.set(number, value)

            return value
        } else {

            return cacheMap.get(number)
        }
    }
}

    方法二：至底向上递推

    递归本身是从上到小，一步步递归到最底层，然后再一步步返回。但是当n很大时，会造成调用栈溢出问题。采用从底向上的方法逐步递推可以解决这个问题

    //通用表达式 f(n)=f(n-1)+f(n-2)
//初始f(1)=1 f(2)=2       f(3)=f(2)+f(1) f(4)=f(3)+f(2)=sum+f(2)
function f2 (n) {
    if (n <= 2) {
        return n
    } else {
        let f1 = 1
        let f2 = 2
        let sum = 0

        for (let i = 3; i <= n; i++) {
            sum = f1 + f2
            f1 = f2
            f2 = sum
        }

        return sum
    }
}

    方法三：尾递归（JavaScript中的一种方法）

    普通递归

     当执行一个函数的时候，就会**创建**一个执行上下文，并且**压入执行上下文栈**，当函数执行完毕的时候，就会将函数的执行上下文从栈中**弹出**。

    ​ 当递归时，JavaScript 会不停的创建执行上下文压入执行上下文栈，对于内存而言，维护这么多的执行上下文开销很大，可以使用尾调用来优化。

    尾调用

    ​ 函数内部的最后一个动作是函数调用。该调用的返回值，直接返回给函数。它的执行栈变化是函数调用时压入执行栈则立马弹出，而普通递归调用是一步步压入栈再一步步弹出

    尾递归

    ​ 函数调用自身，称为递归。如果尾调用自身，就称为尾递归

    //例题1优化
function factorial(n, res) {
 if (n == 1) return res;
 return factorial(n - 1, n * res) //递归时返回的只有函数调用
}

console.log(factorial(4, 1)) // 24

    1.2 动态规划

    含义：记住过往，用变量存储计算过的值，避免后续重复计算；将规模大的问题拆分成几个规模小的问题，然后用小问题推导出大问题，即从底往上

    • 步骤1：用于存储的变量

      不管是用数组还是对象存储数据，得知道保存下来的值代表什么含义。含义一般是题目所求的结果。

    • 步骤2：将大问题拆分为小问题

      假如我们要计算dp[n]，那么需要找出它和dp[n-1]、dp[n-2]……之间存在什么关系

    • 步骤3：找出初始值

      第2步里相当于得到了通用的数学表达式，只有找到初始值才能一步步地计算结果。初始值需要结合步骤2的数学关系式和实际问题

    「例题1」

    一只青蛙一次可以跳上1级台阶，也可以跳上2级。求该青蛙跳上一个n级的台阶总共有多少种跳法

    function f4 (n) {

    let dp = [] //dp[i]:有i级台阶总共有dp[i]种跳法
    
    //初始值
    dp[0] = 0
    dp[1] = 1
    dp[2] = 2

    for (let i = 3; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2] //关系式
    }

    return dp[n]
}

    「例题2」

    一个机器人位于一个 m x n 网格的左上角 （起始点在下图中标记为“Start” ）。机器人每次只能向下或者向右移动一步。机器人试图达到网格的右下角（在下图中标记为“Finish”）。问总共有多少条不同的路径？

    //变量dp[i][j]:表示从左上角走到第i+1行j+1列的所有路径
//关系式dp[i][j]=dp[i][j-1]+dp[i-1][j]
//初始值：从关系式可知i和j中任何一个为0，关系式都不成立，并且通过分析可知它们的值始终为1

function f (m, n) {
    let dp = []


    for (let i = 0; i < m; i++) {
        dp[i] = []
        dp[i][0] = 1
    }

    for (let j = 0; j < n; j++) {
        dp[0][j] = 1
    }

    for (let i = 1; i < m; i++) {
        for (let j = 1; j < n; j++) {
            dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
        }
    }

    return dp[m - 1][n - 1]

}

    「例题3」

    给定一个包含非负整数的 m x n 网格，请找出一条从左上角到右下角的路径，使得路径上的数字总和为最小。输入：二维数组，输出最小数字和

    //变量dp[i][j]:表示从左上角走到第i+1行j+1列的最小路径和
//关系式dp[i][j]=min(dp[i-1][j],dp[i][j-1])+arr[i][j] j>=1,i>=1
//初始值：i和j为0时关系式不满足
//dp[0][j]=dp[0][j-1]+arr[0][j] j>=1,i=0
//dp[i][0]=dp[i-1][0]+arr[i][0] i>=1,j=0
//dp[0][0]=arr[0][0] i=j=0

function f (arr) {
    let dp = []
    let row = arr.length
    let col = arr[0].length

    for (let i = 0; i < row; i++) {
        dp[i] = []
    }

    dp[0][0] = arr[0][0]

    for (let i = 1; i < row; i++) {
        dp[i][0] = dp[i - 1] + arr[i][0]
    }

    for (let j = 1; j < col; j++) {
        dp[0][j] = dp[0][j - 1] + arr[0][j]
    }

    for (let i = 1; i < row; i++) {
        for (let j = 1; j < col; j++) {
            dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1]) + arr[i][j]
        }
    }
    
    if(row===0&&col===0){
        return dp[0][0]
    }else if(row===0){
        return dp[0][col-1]
    }else if(col===0){
        return dp[row-1][0]
    }else{
        return dp[row - 1][col - 1]
    }
}

    「例题4」

    给定两个单词 word1 和 word2，计算出将 word1 转换成 word2 所使用的最少操作数 。你可以对一个单词进行如下三种操作：插入一个字符、删除一个字符、替换一个字符

    //变量dp[i][j]:将长度为i的word1转换为长度为j的word2使用的最少操作数
/*关系式推导
word1和word2完全相等：dp[i][j]=dp[i-1][j-1]+0 即最后一个字符是不需要操作的
word1和word2不等：
1.在word1的末尾插入1个字符：dp[i][j]=dp[i][j-1]+1
2.在word1的末尾删除1个字符：dp[i][j]=dp[i-1][j]+1
3.将word1[i]替换为word2[j]：dp[i][j]=dp[i-1][j-1]+1

dp[i][j]=min(dp[i][j-1],dp[i-1][j],dp[i-1][j-1])+1 i>=1,j>=1
*/
//初始值：以上关系式在i=0或者j=0时不成立，可以知道i=0或j=0可以作为初始值
//dp[0][j]=dp[0][j-1]+1 不停执行插入操作 j>=1
//dp[i][0]=dp[i-1][0]+1 不停执行删除操作 i>=1
//dp[0][0]=0

function f (word1, word2) {
    let n1 = word1.length
    let n2 = word2.length

    let dp = []
    for (let i = 0; i <= n1; i++) { //注意这里取了等号
        dp[i] = []
    }

    dp[0][0] = 0

    for (let i = 1; i <= n1; i++) {//注意这里取了等号
        dp[i][0] = dp[i - 1][0] + 1
    }

    for (let j = 1; j <= n2; j++) {//注意这里取了等号
        dp[0][j] = dp[0][j - 1] + 1
    }

    for (let i = 1; i <= n1; i++) {//注意这里取了等号
        for (let j = 1; j <= n2; j++) {
            if (word1.charAt(i - 1) === word2.charAt(j - 1)) {
                dp[i][j] = dp[i - 1][j - 1] + 0
            }
            else {
                dp[i][j] = Math.min(dp[i - 1][j - 1], dp[i - 1][j], dp[i][j - 1]) + 1
            }

        }
    }

    return dp[n1][n2]
}

    总结：

    • 动态规划大部分都是采用二维数组存储变量
    • 90%的字符串问题可以使用动态规划解决，并且都是二维数组
    • 二维数组存储时，dp[i][j]和 dp[i-1][j] 、dp[i][j-1]、 dp[i-1][j-1]中的某几个肯定存在某种关系，初始值一般就是i=0、j=0、以及i=0 j=0的时候
    • 以上二维数组的动态规划空间复杂度是O(n*m)，后续可以优化为O(min(n,m))

    2.算法技巧总结

    2.1 巧用数组下标

    数组的下标是一个隐含的很有用的数组，特别是在统计一些数字，或者判断一些整型数是否出现过的时候。可以将数字作为数组的下标，数组保存的值为其出现的次数

    「例题」

    有一个无序的int整型数组arr，保存了n个数，并且每个整数的取值范围都在0-20之间，要你在 O(n) 的时间复杂度中把这 n 个数按照从小到大的顺序打印出来。（时间复杂度为 O(N)，空间复杂度为 O(N)）

    function f (arr) {
    let temp = []
    //初始化数组
    for (let i = 0; i < 21; i++) {
        temp[i] = 0
    }

    for (let i = 0; i < arr.length; i++) {
        temp[arr[i]]++ //保存arr[i]这个数字出现的次数
    }

    for (let i = 0; i < 21; i++) {
        for (let j = 0; j < temp[i]; j++) {//数字i出现的次数
            console.log(i)
        }
    }
}

    在遍历数组的时候，会进行越界判断，如果下标差不多要越界了，我们就把它置为0重新遍历。可以用取余数来简化代码

    2.3 巧用双指针

    对于双指针，在做关于单链表的题是特别有用

    2.3.1 判断单链表是否有环

    可以设置一个慢指针和一个快指针来遍历这个链表。慢指针一次移动一个节点，而快指针一次移动两个节点，如果该链表没有环，则快指针会先遍历完这个表，如果有环，则快指针会在第二次遍历时和慢指针相遇。

    2.3.2 如何一次遍历就找到链表中间位置节点

    一样是设置一个快指针和慢指针。慢的一次移动一个节点，而快的两个。在遍历链表的时候，当快指针遍历完成时，慢指针刚好达到中点。

    2.3.3 单链表中倒数第 k 个节点

    设置两个指针，其中一个指针先移动k个节点。之后两个指针以相同速度移动。当那个先移动的指针遍历完成的时候，第二个指针正好处于倒数第k个节点。

n / 4 等价于 n >> 2

n / 8 等价于 n >> 3

    「例题3」

    有一组存放 ID 的数据。并且 ID 取值为 0 - (N-1) 之间，其中只有一个 ID 出现的次数为 1，其他的 ID 出现的次数都等于 2，问如何找到这个次数为 1 的 ID ? （时间复杂度为O(N)，空间复杂度O(1)）

    //根据异或相同为0不同为1，出现两次的ID异或后为0，则所有ID异或后就是出现次数为1的那个数
function f(arr){
    let result=arr[0]
    for(let i=1;i<arr.length;i++){
       result=result^arr[i]
    }
    return result
}

    「例题4」

    假如有 2 个 ID 出现的次数为 1，其他 ID 出现的次数都为 2 呢？

    //出现次数为1的2个ID分别为 A和B，所有ID异或后的结果为A^B，那怎么知道A是多少，B是多少

    let decimalTemp = addZero(maxLength, temp.toString(2))
    let index = decimalTemp.indexOf("1")

    let arr1 = []
    let arr2 = []

    for (let i = 0; i < arr.length; i++) {
        if (addZero(maxLength, arr[i].toString(2))[index] === "1") {
            arr1.push(arr[i])
        } else {
            arr2.push(arr[i])
        }
    }

    let A = arr1[0], B = arr2[0]

    for (let i = 1; i < arr1.length; i++) {
        A = A ^ arr1[i]
    }

    for (let i = 1; i < arr2.length; i++) {
        B = B ^ arr2[i]
    }

    function addZero (maxLength, str) {
        if (str.length < maxLength) {
            let n = maxLength - str.length
            for (let i = 0; i < n; i++) {
                str = "0" + str
            }
        }
        return str
    }

    return [A, B]
}

    「题目1」:一只青蛙一次可以跳上1级台阶，也可以跳上2级。求该青蛙跳上一个n级的台阶总共有多少种跳法

    //方法一：普通递归求解，但是这种递归会重复计算很多相同的值

function f0 (n) {
    if (n <= 2) {
        return n
    } else {
        return f0(n - 1) + f0(n - 2)
    }
}

//优化方法：将每次的递归值存储下来，有则直接返回。 
//这个比原作者的优化方法更好，函数调用只有n次
let cacheMap = new Map()

function f1 (n) {

    if (n <= 2) {
        return n
    } else {

        let value1 = cache(cacheMap, n - 1)
        let value2 = cache(cacheMap, n - 2)
        return value1 + value2
    }

    function cache (cacheMap, number) {
        if (!cacheMap.has(number)) {

            let value = f1(number)
            cacheMap.set(number, value)

            return value
        } else {

            return cacheMap.get(number)
        }
    }
}

    //解法2，采用表达式从上往下嵌套来实现，类似于斐波那契数列
//通用表达式 f(n)=f(n-1)+f(n-2)
//初始f(1)=1 f(2)=2       f(3)=f(2)+f(1) f(4)=f(3)+f(2)=sum+f(2)
function f2 (n) {
    if (n <= 2) {
        return n
    } else {
        let f1 = 1
        let f2 = 2
        let sum = 0

        for (let i = 3; i <= n; i++) {
            sum = f1 + f2
            f1 = f2
            f2 = sum
        }

        return sum
    }
}

    //方法3:动态规划
function f4 (n) {

    let dp = []
    dp[0] = 0
    dp[1] = 1
    dp[2] = 2

    for (let i = 3; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2]
    }

    return dp[n]
}
